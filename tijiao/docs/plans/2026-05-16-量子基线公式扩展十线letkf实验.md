def estimate_local_hyperparams(innovation, y_pert, base_obs_var, base_infl):
    innov_var = float(np.mean(innovation ** 2))
    spread = float(np.mean(y_pert ** 2))
    obs_var_eff = np.clip(0.5 * base_obs_var + 0.5 * innov_var, 0.05, 4.0)

    loc_mix = float(np.clip(spread / (spread + innov_var + 1e-8), 0.05, 0.95))
    return obs_var_eff, infl_eff, loc_mix

def build_prior_metric(y_pert, similarity, lam):
    corr = project_to_psd_correlation(similarity)
    gram = y_pert.T @ y_pert
    gram = 0.5 * (gram + gram.T)
    gram /= np.trace(gram) / max(gram.shape[0], 1)
    metric = (1.0 - lam) * np.eye(gram.shape[0]) + lam * gram
    return project_spd(metric, eps=1e-6)
```

def solve_kkt(c_mat, rhs, a_c, b_c):
    top = np.concatenate([c_mat, a_c.T], axis=1)
    bottom = np.concatenate([a_c, np.zeros((a_c.shape[0], a_c.shape[0]))], axis=1)
    system = np.concatenate([top, bottom], axis=0)
    target = np.concatenate([rhs, b_c], axis=0)
    return stable_inverse(system) @ target
```
